# EJERCICIO 11

función obtenerTipoPatrón():
imprimir("Tipo de patrón [rectángulo, triángulo, rombo]:")
retornar leerEntradaUsuario()

función obtenerTamaño():
imprimir("Tamaño del patrón:")
retornar int(leerEntradaUsuario())

función obtenerCarácter():
imprimir("Carácter para el patrón:")
retornar leerEntradaUsuario()

función generarLínea(tamaño, carácter, relleno):
si relleno:
retornar carácter * tamaño
retornar carácter + " " * (tamaño - 2) + carácter si tamaño > 1 sino carácter

función crearRectángulo(tamaño, carácter):
para i de 1 a tamaño:
imprimir(generarLínea(tamaño, carácter, verdadero))

función crearTriángulo(tamaño, carácter):
para i de 1 a tamaño + 1:
imprimir(carácter * i)

función crearRombo(tamaño, carácter):
para i de 1 a tamaño:
imprimir(" " * (tamaño - i) + carácter * (2 * i - 1))
para i de tamaño - 1 a 1 paso -1:
imprimir(" " * (tamaño - i) + carácter * (2 * i - 1))

función crearPatrón(tipo, tamaño, carácter):
si tipo == "rectángulo":
crearRectángulo(tamaño, carácter)
sino si tipo == "triángulo":
crearTriángulo(tamaño, carácter)
sino si tipo == "rombo":
crearRombo(tamaño, carácter)

tipo = obtenerTipoPatrón()
tamaño = obtenerTamaño()
carácter = obtenerCarácter()
crearPatrón(tipo, tamaño, carácter)
