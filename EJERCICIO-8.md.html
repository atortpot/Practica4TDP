# EJERCICIO 8

INICIO

ESTRUCTURA Cita:
autor
cita
etiquetas // Lista de etiquetas relacionadas con el estado de ánimo o contexto

BASE_DE_DATOS_CITAS = [
Cita("Autor 1", "Cita inspiradora 1", ["motivación", "superación"]),
Cita("Autor 2", "Cita inspiradora 2", ["reflexión", "calma"]),
Cita("Autor 3", "Cita inspiradora 3", ["felicidad", "alegría"]),
// Agregar más citas según sea necesario
]

FUNCIÓN obtenerCitaAleatoria(etiquetas):
citasFiltradas = FILTRAR BASE_DE_DATOS_CITAS POR etiquetas
SI LONGITUD DE citasFiltradas ES MAYOR QUE 0:
índiceAleatorio = NÚMERO ALEATORIO ENTRE 0 Y (LONGITUD DE citasFiltradas - 1)
RETORNAR citasFiltradas[índiceAleatorio]
SINO:
RETORNAR "Lo siento, no tengo una cita para ese estado de ánimo o contexto."

FUNCIÓN solicitarEstadoDeÁnimo():
IMPRIMIR "¿Cómo te sientes hoy? (motivación, reflexión, felicidad, etc.)"
LEER estadoDeÁnimo
RETORNAR estadoDeÁnimo

FUNCIÓN imprimirCita(cita):
SI cita ES UNA INSTANCIA DE Cita:
IMPRIMIR "Cita Inspiradora: " + cita.cita + " - " + cita.autor
SINO:
IMPRIMIR cita // Mensaje de no disponibilidad

FUNCIÓN esperarSolicitudNuevaCita():
IMPRIMIR "¿Deseas otra cita? (sí/no)"
LEER respuesta
RETORNAR respuesta ES IGUAL A "sí"

FUNCIÓN generarCita():
estadoDeÁnimo = solicitarEstadoDeÁnimo()
etiquetas = [estadoDeÁnimo] // Se puede expandir para aceptar múltiples etiquetas
cita = obtenerCitaAleatoria(etiquetas)
imprimirCita(cita)

MIENTRAS VERDADERO:
generarCita()
si NO esperarSolicitudNuevaCita():
SALIR

FIN
