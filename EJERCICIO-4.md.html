# EJERCICIO 4

INICIO

FUNCIÓN obtenerSolicitudUsuario():
IMPRIMIR "Usuario: ¿Cómo puedo ayudarte?"
LEER solicitudUsuario
RETORNAR solicitudUsuario

FUNCIÓN procesarSolicitud(solicitud):
SI solicitud CONTIENE "recordatorio":
RETORNAR "Recordatorio establecido para " + detalles del recordatorio
SINO SI solicitud CONTIENE "buscar":
RETORNAR "Resultados de búsqueda para " + términos de búsqueda
SINO:
RETORNAR "No estoy seguro de cómo ayudar con eso. ¿Puedes ser más específico?"

FUNCIÓN interactuarConUsuario():
solicitud = obtenerSolicitudUsuario()
respuesta = procesarSolicitud(solicitud)
IMPRIMIR "Asistente: " + respuesta

MIENTRAS VERDADERO:
interactuarConUsuario()

FIN
