# EJERCICIO 3

INICIO

DEFINIR API_KEY COMO "tu_api_key_aquí"

FUNCIÓN prepararPrompt(datos):
prompt = "Identifica los principales temas, el sentimiento asociado a cada tema y sugiere tendencias emergentes basadas en el siguiente texto: " + datos
RETORNAR prompt

FUNCIÓN enviarSolicitudAnalisis(prompt):
DEFINIR url COMO "https://api.openai.com/v1/completions"
DEFINIR headers COMO Diccionario con "Content-Type" -> "application/json" y "Authorization" -> "Bearer " + API_KEY
DEFINIR data COMO Diccionario con configuraciones para la solicitud (modelo, prompt, temperatura, max_tokens, top_p, frequency_penalty, presence_penalty)

    HACER solicitud POST a url CON data y headers
    SI respuesta es exitosa:
      RETORNAR texto de la primera opción de la respuesta
    SINO:
      LANZAR excepción "Error en la solicitud a la API"

FUNCIÓN procesarRespuesta(respuesta):
IMPRIMIR "Respuesta de la API:" + respuesta

FUNCIÓN analizarTendencias(datos):
analisisPrompt = prepararPrompt(datos)
respuesta = enviarSolicitudAnalisis(analisisPrompt)
procesarRespuesta(respuesta)

FUNCIÓN obtenerDatosRedesSociales():
RETORNAR "Aquí el texto recogido de redes sociales para analizar"

datos = obtenerDatosRedesSociales()
analizarTendencias(datos)

FIN
