# EJERCICIO 5

INICIO

ESTRUCTURA DispositivoIoT:
id
tipo
ubicación

FUNCIÓN inicializarDispositivos():
dispositivos = obtenerDispositivosIoT()
PARA CADA dispositivo EN dispositivos:
CONFIGURAR dispositivo SEGÚN tipo y ubicación
RETORNAR dispositivos

FUNCIÓN recolectarDatos(dispositivos):
DEFINIR datosComo Lista vacía
PARA CADA dispositivo EN dispositivos:
INTENTAR:
LEER temperatura, humedad, calidadDelAire DE dispositivo
AGREGAR {id: dispositivo.id, temperatura, humedad, calidadDelAire, timestamp: AHORA} A datos
EXCEPTO error DE lectura:
REGISTRAR "Error al leer datos de dispositivo " + dispositivo.id + ": " + error
RETORNAR datos

FUNCIÓN almacenarDatos(datos):
PARA CADA dato EN datos:
GUARDAR dato EN baseDeDatos

FUNCIÓN actualizarInterfazUsuario(datos):
LIMPIAR interfazUsuario
PARA CADA dato EN datos:
MOSTRAR "ID: " + dato.id + ", Temp: " + dato.temperatura + ", Hum: " + dato.humedad + ", Calidad aire: " + dato.calidadDelAire + ", Hora: " + dato.timestamp

FUNCIÓN monitorearAmbiente(dispositivos):
MIENTRAS VERDADERO:
datos = recolectarDatos(dispositivos)
SI datos NO ESTÁ vacío:
almacenarDatos(datos)
actualizarInterfazUsuario(datos)
ESPERAR 5 minutos // Espera antes de la próxima recolección de datos para no sobrecargar los dispositivos ni la interfaz de usuario.

FUNCIÓN obtenerDispositivosIoT():
RETORNAR listaDeDispositivosIoT

dispositivos = inicializarDispositivos()
monitorearAmbiente(dispositivos)

FIN
