función obtenerExpresiónUsuario():
imprimir("Introduce una expresión matemática:")
retornar leerEntradaUsuario()

función esOperadorValido(caracter):
retornar caracter en ['+', '-', '*', '/', '^']

función esParéntesis(caracter):
retornar caracter en ['(', ')']

función comprobarSintaxis(expresión):
para i de 0 a longitud(expresión) - 1:
si no (esDígito(expresión[i]) o esOperadorValido(expresión[i]) o esParéntesis(expresión[i])):
retornar falso
retornar verdadero y expresiónCumpleReglas(expresión)

función expresiónCumpleReglas(expresión):
si longitud(expresión) == 0 o (esOperadorValido(expresión[0]) y expresión[0] != '-') o esOperadorValido(expresión[longitud(expresión)-1]):
retornar falso
para i de 0 a longitud(expresión) - 2:
si esOperadorValido(expresión[i]) y (esOperadorValido(expresión[i+1]) y expresión[i+1] != '-'):
retornar falso
retornar verdadero

función comprobarParentesisBalanceados(expresión):
pila = nuevaPila()
para caracter en expresión:
si caracter == '(':
pila.apilar(caracter)
sino si caracter == ')':
si pila.estaVacia():
retornar falso
pila.desapilar()
retornar pila.estaVacia()

función validarExpresión(expresión):
si comprobarSintaxis(expresión) y comprobarParentesisBalanceados(expresión):
imprimir("La expresión es válida")
sino:
imprimir("La expresión no es válida")

expresión = obtenerExpresiónUsuario()
validarExpresión(expresión)
