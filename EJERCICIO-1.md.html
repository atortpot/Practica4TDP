# EJERCICIO 1

INICIAR Programa

FUNCIÓN obtenerComentarios()
RETORNAR lista de comentarios

FUNCIÓN determinarSentimiento(comentario)
SI comentario contiene palabras positivas
RETORNAR "Positivo"
SI comentario contiene palabras negativas
RETORNAR "Negativo"
SINO
RETORNAR "Neutral"

FUNCIÓN analizarComentarios(comentarios)
PARA CADA comentario EN comentarios
sentimiento = determinarSentimiento(comentario)
IMPRIMIR "Comentario: ", comentario, " - Sentimiento: ", sentimiento

PRINCIPAL
comentarios = obtenerComentarios()
analizarComentarios(comentarios)

FIN Programa
